Variables y Tipos de Datos

variables.py 
# Variables en Python
nombre = "Gabriel"      # str (texto)
edad = 25                 # int (entero)
promedio = 85.5          # float (decimal)

print(f"Hola, me llamo {nombre}")
print(f"Tengo {edad} años")
print(f"Mi promedio es {promedio}")
print(f"¿Soy estudiante? {es_estudiante}")

# Tipo de dato
print(type(nombre))         # <class 'str'>
print(type(edad))          # <class 'int'>

Operadores Aritméticos

aritmetica.py 
# Operadores aritméticos
a = 10
b = 3

print(a + b)    # 13 - Suma
print(a - b)    # 7  - Resta
print(a * b)    # 30 - Multiplicación
print(a / b)    # 3.333... - División
print(a // b)   # 3  - División entera
print(a % b)     # 1  - Módulo (resto)
print(a ** b)   # 1000 - Potencia

# Ejemplo práctico
precio = 100
descuento = 20
precio_final = precio - (precio * descuento / 100)
print(precio_final)  # 80.0

Bucle WHILE

while_basico.py

    contador += 1

Break y Continue

while_control.py 
# break - salir del ciclo
while True:
    respuesta = input("¿Salir? (s/n): ")
    if respuesta == "s":
        print("¡Chao!")
        break

# continue - siguiente iteración
for i in range(10):
    if i == 3 or i == 6:
        continue    # Saltar
    print(i)           # 0,1,2,4,5,7,8,9

# Encontrar primer número divisible
for n in range(1, 100):
    if n % 7 == 0:
        print(f"找到: {n}")
        break

Crear Funciones

funciones.py

    print("¡Hola!")

    print(f"¡Hola {nombre}!")

Parámetros y Retorno

parametros.py 
# Parámetros con valor por defecto
def saludar(nombre="Usuario"):
    return f"Hola {nombre}"

print(saludar())           # Hola Usuario
print(saludar("Ana"))     # Hola Ana

# Múltiples parámetros
def promedio(a, b, c):
    return (a + b + c) / 3

print(promedio(70, 80, 90))  # 80.0

# Parámetros opcionales
def info(nombre, edad=None):
    if edad:
        return f"{nombre}, {edad} años"
    return nombre

print(info("Gabriel", 25))

Raise y Excepciones Personalizadas

raise.py 
# Lanzar excepciones
def dividir(a, b):
    if b == 0:
        raise ValueError("Divisor no puede ser 0")
    return a / b

# Excepción personalizada
class MiError(Exception):
    pass

def validar(edad):
    if edad < 0:
        raise MiError("Edad negativa")

# Try inside try
try:
    try:
        x = validar(-5)
    except MiError as e:
        print(f"Error personalizado: {e}")
except Exception:
    pass

Leer y Escribir Archivos

archivos.py 
# Escribir archivo
with open("archivo.txt", "w") as f:
    f.write("Hola mundo\n")
    f.write("Segunda línea")

# Leer archivo
with open("archivo.txt", "r") as f:
    contenido = f.read()
    print(contenido)

# Leer línea por línea
with open("archivo.txt", "r") as f:
    for linea in f:
        print(linea.strip())
