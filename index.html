🔄 Bucle FOR

for.py 
# Range
for i in range(5):
    print(i)  # 0,1,2,3,4

# Con paso
for i in range(0, 10, 2):
    print(i)  # 0,2,4,6,8

# Iterar lista
nombres = ["Rick", "Morty", "Summer"]
for nombre in nombres:
    print(f"¡Hola {nombre}!")

🛑 Control

break_continue.py 
# Continue - salta
for i in range(5):
    if i == 2: continue
    print(i)  # 0,1,3,4

# Break - sale
for i in range(10):
    if i == 3: break
    print(i)  # 0,1,2

👾 Clases

clase.py 
class Persona:
    def __init__(self, nombre, edad):
        self.nombre = nombre
        self.edad = edad
    
    def saludar(self):
        return f"¡Hola, soy {self.nombre}!"

persona = Persona("Rick", 65)
print(persona.saludar())

🧬 Herencia

herencia.py 
class Ser:
    def __init__(self, nombre):
        self.nombre = nombre

class Humano(Ser):
    def decir_hola(self):
        return "¡Hola!"

rick = Humano("Rick")
print(rick.decir_hola())
