Logicos

logicos.py 
edad = 20
tiene_entrada = True

# and - ambas condiciones
if edad >= 18 and tiene_entrada:
    print("Puedes entrar")

# or - al menos una
if edad < 18 or not tiene_entrada:
    print("No puedes entrar")

Bucle FOR

for.py 
# Range basico
for i in range(5):
    print(i)  # 0,1,2,3,4

# Con paso
for i in range(0, 10, 2):
    print(i)  # 0,2,4,6,8

# Iterar lista
nombres = ["Ana", "Luis", "Maria"]
for nombre in nombres:
    print(f"Hola {nombre}")

Control

break_continue.py 
# continue - salta iteracion
for i in range(5):
    if i == 2: continue
    print(i)  # 0,1,3,4

# break - sale del bucle
for i in range(10):
    if i == 3: break
    print(i)  # 0,1,2

Clases

clase.py 
class Persona:
    def __init__(self, nombre, edad):
        self.nombre = nombre
        self.edad = edad
    
    def saludar(self):
        return f"Hola, soy {self.nombre}"

persona = Persona("Carlos", 25)
print(persona.saludar())

Herencia

herencia.py 
class SerVivo:
    def __init__(self, nombre):
        self.nombre = nombre

class Humano(SerVivo):
    def hablar(self):
        return "Hola!"

humano = Humano("Carlos")
print(humano.hablar())
