Variables y Tipos

variables.py
# Variables en Python
nombre = "Gabriel"      # str
edad = 25               # int
promedio = 85.5         # float

print(f"Tengo {edad} años")

Conversión de Tipos

conversion.py
# Conversión de tipos
numero = int("42")      # String -> int
precio = float("19.99") # String -> float
edad = 25
print("Tengo " + str(edad) + " años")

print(int(False))       # 0

Operadores Aritméticos

aritmetica.py
a, b = 10, 3
print(a + b)    # 13 Suma
print(a - b)    # 7  Resta
print(a * b)    # 30 Multiplicación
print(a / b)    # 3.33...
print(a // b)   # 3  División entera
print(a % b)     # 1  Módulo
print(a ** b)   # 1000 Potencia

Comparación

comparacion.py

print(a == b)   # False
print(a != b)   # True
print(a < b)    # True
print(a > b)    # False

if a < b:
    print("a es menor que b")

Condicional if/else

if_simple.py
edad = 18

if edad >= 18:
    print("Mayor de edad")
else:
    print("Menor de edad")

# Multiple
nota = 75
if nota >= 70:
    print("Acreditado")
elif nota >= 50:
    print("Aceptable")
else:
    print("Reprobado")

Operadores Lógicos

logicos.py

# and, or, not

    print("Puede conducir")

if edad < 18 or not tiene_licencia:

Bucle FOR

for_basico.py
# range
for i in range(5):
    print(i)  # 0,1,2,3,4

# Con inicio y paso
for i in range(0, 10, 2):
    print(i)  # 0,2,4,6,8

# Iterar lista
nombres = ["Ana", "Bob", "Carla"]
for nombre in nombres:
    print(f"Hola {nombre}")

enumerate

enumerate.py

# Con índice
for i, fruta in enumerate(frutas):
    print(f"{i}: {fruta}")

# Sumar con for
numeros = [10, 20, 30]

Bucle WHILE

while_basico.py
contador = 0
while contador < 5:
    print(contador)
    contador += 1

# Con break
while True:
    print("Ejecutando...")
    break

Break y Continue

control.py
# Continue - salta iteración
for i in range(5):
    if i == 2:
        continue
    print(i)  # 0,1,3,4

# Break - sale del ciclo
for i in range(10):
    if i == 3:
        break
    print(i)  # 0,1,2

Crear Funciones

funciones.py
def saludar(nombre):
    return f"Hola {nombre}!"

print(saludar("Gabriel"))

# Con valor por defecto
def saludar(nombre="Usuario"):
    return f"Hola {nombre}"

print(saludar())

Retorno

retorno.py
def promedio(a, b, c):
    return (a + b + c) / 3

resultado = promedio(70, 80, 90)
print(f"Promedio: {resultado}")

# Múltiples valores
def operaciones(a, b):
    return a + b, a - b

suma, resta = operaciones(10, 5)

Métodos de Listas

metodos.py
numeros = [3, 1, 4, 1, 5]
numeros.sort()
print(numeros)  # [1,1,3,4,5]

numeros.reverse()
print(numeros)  # [5,4,3,1,1]

print(numeros.count(1))  # 2
print(numeros.index(3))  # 2
print(numeros[1:4])      # [4,1,5]

Try/Except

excepciones.py
try:
    resultado = 10 / 0
except ZeroDivisionError:
    print("No se puede dividir por cero")
except Exception as e:
    print(f"Error: {e}")
finally:
    print("Siempre se ejecuta")

Raise

raise.py
def dividir(a, b):
    if b == 0:
        raise ValueError("Divisor no puede ser 0")
    return a / b

try:
    print(dividir(10, 0))
except ValueError as e:
    print(f"Error: {e}")
