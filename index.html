Variables y Tipos de Datos

variables.py 
# Variables en Python
nombre = "Gabriel"      # str (texto)
edad = 25                 # int (entero)
promedio = 85.5          # float (decimal)
es_estudiante = True      # bool (booleano)

# Mostrar valores
print(f"Hola, me llamo {nombre}")
print(f"Tengo {edad} años")
print(f"Mi promedio es {promedio}")
print(f"¿Soy estudiante? {es_estudiante}")

# Tipo de dato
print(type(nombre))         # <class 'str'>
print(type(edad))          # <class 'int'>

Conversión de Tipos

conversion.py 
# Conversión de tipos

valor = float("19.99")

Operadores Aritméticos

aritmetica.py 
# Operadores aritméticos
a = 10
b = 3

print(a + b)    # 13 - Suma
print(a - b)    # 7  - Resta
print(a * b)    # 30 - Multiplicación
print(a / b)    # 3.333... - División
print(a // b)   # 3  - División entera
print(a % b)     # 1  - Módulo (resto)
print(a ** b)   # 1000 - Potencia

# Ejemplo práctico
precio = 100
descuento = 20
precio_final = precio - (precio * descuento / 100)
print(precio_final)  # 80.0

Operadores de Comparación

comparacion.py 
# Operadores de comparación
a = 5
b = 10

print(a == b)   # False - Igual a
print(a != b)   # True  - Diferente de
print(a < b)    # True  - Menor que
print(a > b)    # False - Mayor que
print(a <= b)   # True  - Menor o igual
print(a >= b)   # False - Mayor o igual

# Usos comunes
edad = 18
if edad >= 18:
    print("Puede votar")

puntaje = 75
if puntaje >= 70:
    print("Aprobado")

Bucle FOR Básico

for_basico.py 
# range(inicio, fin, paso)
for i in range(5):
    print(i)           # 0,1,2,3,4

for i in range(1, 6):
    print(i)           # 1,2,3,4,5

for i in range(0, 10, 2):
    print(i)           # 0,2,4,6,8

# Iterar sobre lista
nombres = ["Ana", "Bob", "Carla"]
for nombre in nombres:
    print(f"Hola {nombre}")

FOR con Funciones Útiles

# enumerate - obtener índice
frutas = ["manzana", "pera", "uva"]

Bucle WHILE

while_basico.py 
# While básico
contador = 0
while contador < 5:
    print(contador)
    contador += 1

# Con input
intento = 0
while intento < 3:
    clave = input("Ingresa clave: ")
    if clave == "secreto":
        print("¡Correcto!")
        break
    intento += 1
else:
    print("Intentos agotados")

# Validar entrada
numero = 0
while numero < 1 or numero > 10:
    numero = int(input("Número 1-10: "))

Break y Continue

while_control.py 
# break - salir del ciclo
while True:
    respuesta = input("¿Salir? (s/n): ")
    if respuesta == "s":
        print("¡Chao!")
        break

# continue - siguiente iteración
for i in range(10):
    if i == 3 or i == 6:
        continue    # Saltar
    print(i)           # 0,1,2,4,5,7,8,9

# Encontrar primer número divisible
for n in range(1, 100):
    if n % 7 == 0:
        print(f"找到: {n}")
        break

Crear Funciones

funciones.py 
# Función sin parámetros
def saludar():
    print("¡Hola!")

saludar()

# Función con parámetro
def saludar(nombre):
    print(f"¡Hola {nombre}!")

saludar("Gabriel")

# Función con retorno
def sumar(a, b):
    return a + b

resultado = sumar(3, 5)
print(resultado)          # 8

Parámetros y Retorno

parametros.py 
# Parámetros con valor por defecto
def saludar(nombre="Usuario"):
    return f"Hola {nombre}"

print(saludar())           # Hola Usuario
print(saludar("Ana"))     # Hola Ana

# Múltiples parámetros
def promedio(a, b, c):
    return (a + b + c) / 3

print(promedio(70, 80, 90))  # 80.0

# Parámetros opcionales
def info(nombre, edad=None):
    if edad:
        return f"{nombre}, {edad} años"
    return nombre

print(info("Gabriel", 25))

Operaciones con Listas

listas.py 
# Crear lista
frutas = ["manzana", "pera", "uva"]

# Acceder elementos
print(frutas[0])         # manzana
print(frutas[-1])        # uva

# Modificar
frutas[0] = "naranja"

# Agregar
frutas.append("coco")    # Al final
frutas.insert(1, "melón") # En posición

# Eliminar
frutas.pop()           # Elimina último
frutas.remove("pera")   # Por valor

# Longitud
print(len(frutas))

Métodos de Listas

lista_metodos.py 
# Ordenar
numeros = [3, 1, 4, 1, 5]
numeros.sort()           # [1,1,3,4,5]
numeros.sort(reverse=True)

# Invertir
numeros.reverse()

# Contar
print(numeros.count(1))

# Índice
print(numeros.index(3))

# Slicing (rebanar)
lista = [0,1,2,3,4,5]
print(lista[1:4])        # [1,2,3]
print(lista[:3])         # [0,1,2]
print(lista[3:])         # [3,4,5]

Diccionarios

diccionarios.py 
# Crear diccionario
persona = {
    "nombre": "Gabriel",
    "edad": 25,
    "ciudad": "Santiago"
}

# Acceder valores
print(persona["nombre"])

# .get() (seguro)
print(persona.get("edad"))

# Agregar/modificar
persona["edad"] = 26
persona["email"] = "correo@..."

# Eliminar
persona.pop("ciudad")
# o
del persona["edad"]

Métodos de Diccionarios

dicc_metodos.py 
# Keys, values, items
persona = {"nombre": "Ana", "edad": 25}

print(persona.keys())    # keys
print(persona.values())  # values
print(persona.items())  # pares

# Iterar
for clave in persona:
    print(clave, persona[clave])

for clave, valor in persona.items():
    print(f"{clave}: {valor}")

# Copiar
copia = persona.copy()
