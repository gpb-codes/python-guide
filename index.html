Break y Continue

control.py 
# Continue - salta iteración
for i in range(5):
    if i == 2: continue
    print(i)  # 0,1,3,4

# Break - sale del ciclo
for i in range(10):
    if i == 3: break
    print(i)  # 0,1,2

Raise

raise.py 
def dividir(a, b):
    if b == 0:
        raise ValueError("Divisor no puede ser 0")
    return a / b

try:
    print(dividir(10, 0))
except ValueError as e:
    print(f"Error: {e}")

Clases y Objetos

clase.py 
class Persona:
    def __init__(self, nombre, edad):
        self.nombre = nombre
        self.edad = edad
    
    def saludar(self):
        return f"Hola, soy {self.nombre}"
    
    def cumplir_anios(self):
        self.edad += 1

persona = Persona("Gabriel", 25)
print(persona.saludar())
